<?php
declare(strict_types=1);

namespace App\Controller;

use Hyperf\Contract\OnCloseInterface;
use Hyperf\Contract\OnMessageInterface;
use Hyperf\Contract\OnOpenInterface;
use Hyperf\Utils\ApplicationContext;
use Swoole\Http\Request;
use Swoole\Server;
use Swoole\Websocket\Frame;
use Swoole\WebSocket\Server as WebSocketServer;

class WebSocketController implements OnMessageInterface, OnOpenInterface, OnCloseInterface
{

    protected $key;

    protected $ttl = 7200;

    protected $container;
    protected $redis;

    public function __construct()
    {
        $this->container = ApplicationContext::getContainer();
        $this->redis = $this->container->get(\Hyperf\Redis\Redis::class);
        $this->key = config('rdkey.ws_visited_client');
    }

    /**
     * еҸ‘йҖҒж¶ҲжҒҜ
     * @param \Swoole\Http\Response|WebSocketServer $server
     * @param Frame $frame
     */
    public function onMessage($server, Frame $frame): void
    {
        $fdList = $this->redis->sMembers($this->key);
        //еҰӮжһңеҪ“еүҚе®ўжҲ·з«ҜеңЁе®ўжҲ·з«ҜйӣҶеҗҲдёӯ,е°ұеҲ·ж–°
        if (in_array($frame->fd, $fdList)) {
            $this->appendFd($frame->fd);
        } else {
            $server->push($frame->fd, 'disconnect');
        }
    }

    /**
     * е®ўжҲ·з«ҜеӨұеҺ»иҝһжҺҘ
     * @param \Swoole\Http\Response|Server $server
     * @param int $fd
     * @param int $reactorId
     */
    public function onClose($server, int $fd, int $reactorId): void
    {
        $this->redis->sRem($this->key, $fd);
        // var_dump('closed');
    }

    /**
     * е®ўжҲ·з«ҜиҝһжҺҘ
     * @param \Swoole\Http\Response|WebSocketServer $server
     * @param Request $request
     */
    public function onOpen($server, Request $request): void
    {
        if ($this->appendFd($request->fd)) {
            $server->push($request->fd, 'Opened Succeed');
        } else {
            $server->push($request->fd, 'Opened Failed');
        }
    }

    /**
     * з»ҙжҠӨе®ўжҲ·з«Ҝ
     * @param $fd
     * @return bool
     */
    private function appendFd($fd)
    {
        return $this->redis->sAdd($this->key, $fd) && $this->redis->expire($this->key, $this->ttl);
    }

}

д»Јз Ғи·Ҝз”ұйңҖиҰҒжҺҘж”¶еҜ№еә”зҡ„ https://domain жҲ– https://domain/socket/

жҺҘзқҖпјҢиҝһеҝҷжӢҝеҹҹеҗҚиҝӣиЎҢеҶҚж¬ЎиҝһжҺҘжөӢиҜ•пјҢз»ҲдәҺзңӢеҲ°дәҶ101 Switching Protocolsзҡ„е“Қеә”Status CodeгҖӮе°ұиҝҷж ·пјҢд№ҹз®—жҳҜз»ҲдәҺи§ЈеҶіе®ҢеңЁ HTTPS дёӢд»Ҙ wss://{еҹҹеҗҚ}/ зҡ„ж–№ејҸиҝһжҺҘ WebSocketзҡ„дёҖзі»еҲ—й—®йўҳгҖӮ

е…ідәҺNginxдёӯзҡ„WebSocketй…ҚзҪ®

вҖғвҖғ иҮӘ1.3 зүҲжң¬ејҖе§ӢпјҢNginxе°ұж”ҜжҢҒ WebSocketпјҢе№¶дё”еҸҜд»Ҙдёә WebSocket еә”з”ЁзЁӢеәҸеҒҡеҸҚеҗ‘д»ЈзҗҶе’ҢиҙҹиҪҪеқҮиЎЎгҖӮWebSocket е’Ң HTTP жҳҜдёӨз§ҚдёҚеҗҢзҡ„еҚҸи®®пјҢдҪҶжҳҜ WebSocket дёӯзҡ„жҸЎжүӢе’Ң HTTP дёӯзҡ„жҸЎжүӢе…је®№пјҢе®ғдҪҝз”Ё HTTP дёӯзҡ„ Upgrade еҚҸи®®еӨҙе°ҶиҝһжҺҘд»Һ HTTP еҚҮзә§еҲ° WebSocketпјҢеҪ“е®ўжҲ·з«ҜеҸ‘иҝҮжқҘдёҖдёӘ Connection: UpgradeиҜ·жұӮеӨҙж—¶пјҢе…¶е®һNginxжҳҜдёҚзҹҘйҒ“зҡ„гҖӮжүҖд»ҘпјҢеҪ“ Nginx д»ЈзҗҶжңҚеҠЎеҷЁжӢҰжҲӘеҲ°дёҖдёӘе®ўжҲ·з«ҜеҸ‘жқҘзҡ„ Upgrade иҜ·жұӮж—¶пјҢйңҖиҰҒжҲ‘д»¬жҳҫејҸзҡ„й…ҚзҪ®ConnectionгҖҒUpgradeеӨҙдҝЎжҒҜпјҢе№¶дҪҝз”Ё 101пјҲдәӨжҚўеҚҸи®®пјүиҝ”еӣһе“Қеә”пјҢеңЁе®ўжҲ·з«ҜгҖҒд»ЈзҗҶжңҚеҠЎеҷЁе’ҢеҗҺз«Ҝеә”з”ЁжңҚеҠЎд№Ӣй—ҙе»әз«Ӣйҡ§йҒ“жқҘж”ҜжҢҒ WebSocketгҖӮ

вҖғвҖғ еҪ“з„¶пјҢиҝҳйңҖиҰҒжіЁж„ҸдёҖзӮ№пјҢжӯӨж—¶WebSocket д»Қз„¶еҸ—еҲ° Nginx зјәзңҒдёә60з§’зҡ„ proxy_read_timeout й…ҚзҪ®еҪұе“ҚгҖӮиҝҷж„Ҹе‘ізқҖпјҢеҰӮжһңдҪ жңүдёҖдёӘзЁӢеәҸдҪҝз”ЁдәҶ WebSocketпјҢдҪҶеҸҲеҸҜиғҪи¶…иҝҮ60з§’дёҚеҸ‘йҖҒд»»дҪ•ж•°жҚ®зҡ„иҜқпјҢйӮЈд№ҲйңҖиҰҒеўһеӨ§и¶…ж—¶ж—¶й—ҙпјҲй…ҚзҪ®proxy_read_timeoutпјүпјҢиҰҒд№Ҳе®һзҺ°дёҖдёӘPingгҖҒPongзҡ„еҝғи·іж¶ҲжҒҜд»ҘдҝқжҢҒе®ўжҲ·з«Ҝе’ҢжңҚеҠЎз«Ҝзҡ„иҒ”зі»гҖӮдҪҝз”ЁPingгҖҒPongзҡ„и§ЈеҶіж–№жі•жңүйўқеӨ–зҡ„еҘҪеӨ„пјҢеҰӮпјҡеҸҜд»ҘеҸ‘зҺ°иҝһжҺҘжҳҜеҗҰиў«ж„ҸеӨ–е…ій—ӯзӯүгҖӮ вҖғвҖғ

е…ідәҺжңҖеҗҺзҡ„иҝҷдёӘе°Ҹй—®йўҳпјҢдё»иҰҒжҳҜеңЁеҜ№Nginxй…ҚзҪ®зҡ„ж—¶еҖҷе°Ҷlocation=/зҡ„иҜ·жұӮйғҪиҝӣиЎҢдәҶproxy_passпјҲиҪ¬еҸ‘пјүгҖӮз”ұдәҺh5е®ўжҲ·з«Ҝзҡ„ж–Үд»¶жү“еҢ…жҲҗйқҷжҖҒж–Үд»¶еҗҺпјҢеӯҳж”ҫеңЁжңҚеҠЎеҷЁзҡ„жҢҮе®ҡзӣ®еҪ•дёӢпјҲиҝҷйҮҢеҒҮи®ҫеңЁ/root/html/static/и·Ҝеҫ„дёӢпјүпјҢиҝҷд№ҹе°ұеҜјиҮҙиҝҷз§Қй…ҚзҪ®зҡ„жғ…еҶөдёӢNginxж— жі•жӯЈеёёд»ЈзҗҶжҢҮе®ҡзӣ®еҪ•дёӢзҡ„е®ўжҲ·з«Ҝж–Үд»¶гҖӮдәҺжҳҜеҶҚдёҖж¬Ўдҝ®ж”№й…ҚзҪ®ж–Үд»¶пјҢж·»еҠ locationй…ҚзҪ®пјҢжңҖз»Ҳи§ЈеҶіжүҖжңүй—®йўҳгҖӮ
